# Запускаем программу из 4-1 в фоне
./task_4_1 &
PID=$!

# Анализируем память
cat /proc/$PID/status | grep -E "VmSize|VmRSS"

# Пример вывода:
# VmSize:   12345 kB
# VmRSS:     5678 kB

# VmSize — виртуальная память (включая неиспользуемые страницы)
# VmRSS — реально используемая физическая память

# Завершаем процесс
kill $PID
